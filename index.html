Notes Taking App

Write, save, and delete notes — all stored in your browser with localStorage.

Source Code

// Notes App — CRUD with edit support

let notes = JSON.parse(localStorage.getItem("notes")) || [];
let editingId = null;

function startEdit(id) {
  editingId = id;
  render();
}

function saveEdit(id) {
  const card = document.querySelector(`[data-id="${id}"]`);
  const title = card.querySelector(".edit-title").value.trim() || "Untitled";
  const text = card.querySelector(".edit-text").value.trim();
  if (!text) return;
  const note = notes.find(n => n.id === id);
  if (note) { note.title = title; note.text = text; }
  editingId = null;
  save();
}

function render() {
  grid.innerHTML = notes.map(n => {
    if (n.id === editingId) {
      return `<div class="note-card" data-id="${n.id}">
        <input class="note-edit-input edit-title" value="${n.title}">
        <textarea class="note-edit-input edit-text">${n.text}</textarea>
        <div class="note-edit-actions">
          <button onclick="saveEdit(${n.id})">Save</button>
          <button onclick="cancelEdit()">Cancel</button>
        </div>
      </div>`;
    }
    return `<div class="note-card" data-id="${n.id}">
      <div class="note-actions">
        <button onclick="startEdit(${n.id})">✎</button>
        <button onclick="deleteNote(${n.id})">✕</button>
      </div>
      <div class="note-title">${n.title}</div>
      <div class="note-text">${n.text}</div>
    </div>`;
  }).join("");
}
